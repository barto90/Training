Step 2: Add PowerShell Function Code

Replace the default function code with this PowerShell script:

# PowerShell script for processing ServiceBus messages
using namespace System.Net

param($QueueItem, $TriggerMetadata)

# Log the incoming message
Write-Host "üöå ServiceBus Message Received!"
Write-Host "üìÖ Timestamp: $(Get-Date)"
Write-Host "üÜî Message ID: $($TriggerMetadata.MessageId)"
Write-Host "üìã Message Content: $QueueItem"
Write-Host "üìä Delivery Count: $($TriggerMetadata.DeliveryCount)"
Write-Host "‚è∞ Enqueued Time: $($TriggerMetadata.EnqueuedTimeUtc)"

# Process the message content
try {
    # Convert message if it's JSON
    if ($QueueItem -match '^[\{\[].*[\}\]]$') {
        $messageObject = $QueueItem | ConvertFrom-Json
        Write-Host "‚úÖ Successfully parsed JSON message:"
        $messageObject | ConvertTo-Json -Depth 3 | Write-Host
    } else {
        Write-Host "üìù Plain text message: $QueueItem"
    }
    
    # Simulate message processing
    Write-Host "‚öôÔ∏è Processing message..."
    Start-Sleep -Seconds 2
    Write-Host "‚úÖ Message processed successfully!"
    
} catch {
    Write-Error "‚ùå Error processing message: $($_.Exception.Message)"
    throw
}

Write-Host "üèÅ Message processing completed."

Step 3: Create Message Publisher Script

Save this PowerShell script to send test messages:

# Send-ServiceBusMessage.ps1
param(
    [Parameter(Mandatory=$true)]
    [string]$ConnectionString,
    
    [Parameter(Mandatory=$true)]
    [string]$QueueName,
    
    [Parameter(Mandatory=$false)]
    [string]$MessageContent = "Hello from PowerShell!"
)

# Install required module if not present
if (!(Get-Module -ListAvailable -Name Azure.ServiceBus.Core)) {
    Install-Module -Name Azure.ServiceBus.Core -Force -AllowClobber
}

Import-Module Azure.ServiceBus.Core

try {
    # Create ServiceBus client
    $serviceBusClient = [Azure.Messaging.ServiceBus.ServiceBusClient]::new($ConnectionString)
    $sender = $serviceBusClient.CreateSender($QueueName)
    
    # Create message
    $message = [Azure.Messaging.ServiceBus.ServiceBusMessage]::new($MessageContent)
    $message.MessageId = [Guid]::NewGuid().ToString()
    
    # Send message
    Write-Host "üì§ Sending message to queue: $QueueName"
    Write-Host "üìã Message content: $MessageContent"
    
    $sender.SendMessageAsync($message).GetAwaiter().GetResult()
    
    Write-Host "‚úÖ Message sent successfully!"
    Write-Host "üÜî Message ID: $($message.MessageId)"
    
} catch {
    Write-Error "‚ùå Failed to send message: $($_.Exception.Message)"
} finally {
    if ($sender) { $sender.DisposeAsync().GetAwaiter().GetResult() }
    if ($serviceBusClient) { $serviceBusClient.DisposeAsync().GetAwaiter().GetResult() }
}
